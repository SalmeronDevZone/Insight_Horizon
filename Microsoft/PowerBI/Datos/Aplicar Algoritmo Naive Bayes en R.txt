# Cargar dataset
 dataset <-read.table("C:\\Demos MI Power BI\\datos\\Clientes_ML.csv",header = TRUE, sep = ",")

# Cargar paquete e1071 
library(e1071)

# Eliminar columnas continuas
dataset$Edad <- NULL
dataset$Ingresos <- NULL

# Modelo Naive Bayes
Modelo_NB <- naiveBayes(dataset[,2:11], dataset[,12])

# Data frame con las predicciones para todas las filas
Predicción_NB <- as.data.frame(predict(Modelo_NB, dataset, type = "raw"))

# Combinar datos originales con la predicción
Clientes_Predicción_NB <- cbind(dataset, Predicción_NB)
